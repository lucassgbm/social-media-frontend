Scripts para criar o runner do node no Play Docker

docker volume create gitlab-runner-config

docker run --detach \
--name gitlab-runner-job \
--restart always \
--memory-reservation 32m \
--memory 64m \
--memory-swap 128m \
--cpus .1 \
--volume /var/run/docker.sock:/var/run/docker.sock \
--volume gitlab-runner-config:/etc/gitlab-runner \
--volume /etc/timezone:/etc/timezone:ro \
--volume /usr/share/zoneinfo:/usr/share/zoneinfo:ro \
gitlab/gitlab-runner:alpine-v17.6.1

docker exec --interactive --privileged --tty gitlab-runner-job bash
